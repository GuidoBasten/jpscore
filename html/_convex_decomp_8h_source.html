136 mark_domains(CDT& ct,
137  CDT::Face_handle start,
138  int index,
139  std::list<CDT::Edge>& border )
140 {
141  if(start->info().nesting_level != -1) {
142  return;
143  }
144  std::list<CDT::Face_handle> queue;
145  queue.push_back(start);
146 
147  while(! queue.empty()) {
148  CDT::Face_handle fh = queue.front();
149  queue.pop_front();
150  if(fh->info().nesting_level == -1) {
151  fh->info().nesting_level = index;
152  for(int i = 0; i < 3; i++) {
153  CDT::Edge e(fh,i);
154  CDT::Face_handle n = fh->neighbor(i);
155  if(n->info().nesting_level == -1) {
156  if(ct.is_constrained(e)) border.push_back(e);
157  else queue.push_back(n);
158  }
159  }
160  }
161  }
162 }

164 //explore set of facets connected with non constrained edges,
165 //and attribute to each such set a nesting level.
166 //We start from facets incident to the infinite vertex, with a nesting
167 //level of 0. Then we recursively consider the non-explored facets incident
168 //to constrained edges bounding the former set and increase the nesting level by 1.
169 //Facets in the domain are those with an odd nesting level.
170 inline
171 void
172 mark_domains(CDT& cdt)
173 {
174  for(CDT::All_faces_iterator it = cdt.all_faces_begin(); it != cdt.all_faces_end(); ++it) {
175  it->info().nesting_level = -1;
176  }
177 
178  int index = 0;
179  std::list<CDT::Edge> border;
180  mark_domains(cdt, cdt.infinite_face(), index++, border);
181  while(! border.empty()) {
182  CDT::Edge e = border.front();
183  border.pop_front();
184  CDT::Face_handle n = e.first->neighbor(e.second);
185  if(n->info().nesting_level == -1) {
186  mark_domains(cdt, n, e.first->info().nesting_level+1, border);
187  }
188  }
189 }
